```bash
npm install --save my-modal-lib-pro
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'my-modal-lib-pro'
import 'my-modal-lib-pro/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
